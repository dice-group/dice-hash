60  inline size_t hash_bytes(void const *ptr, size_t len) noexcept {
61 
62 
63  static constexpr unsigned int r = 47;
64 
65  auto const *const data64 = static_cast<uint64_t const *>(ptr);
66  uint64_t h = seed ^ (len * m);
67 
68  size_t const n_blocks = len / 8;
69  for (size_t i = 0; i < n_blocks; ++i) {
70  auto k = unaligned_load<uint64_t>(data64 + i);
71 
72  k *= m;
73  k ^= k >> r;
74  k *= m;
75 
76  h ^= k;
77  h *= m;
78  }
79 
80  auto const *const data8 = reinterpret_cast<uint8_t const *>(data64 + n_blocks);
81  switch (len & 7U) {
82  case 7:
83  h ^= static_cast<uint64_t>(data8[6]) << 48U;
84  [[fallthrough]];
85  case 6:
86  h ^= static_cast<uint64_t>(data8[5]) << 40U;
87  [[fallthrough]];
88  case 5:
89  h ^= static_cast<uint64_t>(data8[4]) << 32U;
90  [[fallthrough]];
91  case 4:
92  h ^= static_cast<uint64_t>(data8[3]) << 24U;
93  [[fallthrough]];
94  case 3:
95  h ^= static_cast<uint64_t>(data8[2]) << 16U;
96  [[fallthrough]];
97  case 2:
98  h ^= static_cast<uint64_t>(data8[1]) << 8U;
99  [[fallthrough]];
100  case 1:
101  h ^= static_cast<uint64_t>(data8[0]);
102  h *= m;
103  [[fallthrough]];
104  default:
105  break;
106  }
107 
108  h ^= h >> r;
109  h *= m;
110  h ^= h >> r;
111  return static_cast<size_t>(h);
112  }

159  inline size_t hash_int(uint64_t x) noexcept {
160  // inspired by lemire's strongly universal hashing
161  // https://lemire.me/blog/2018/08/15/fast-strongly-universal-64-bit-hashing-everywhere/
162  //
163  // Instead of shifts, we use rotations so we don't lose any bits.
164  //
165  // Added a final multiplication with a constant for more mixing. It is most important that
166  // the lower bits are well mixed.
167  auto h1 = x * 0xA24BAED4963EE407UL;
168  auto h2 = rotr(x, 32U) * 0x9FB21C651E98DF25UL;
169  auto h = rotr(h1 + h2, 32U);
170  return static_cast<size_t>(h);
171  }
